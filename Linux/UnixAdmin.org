* Sudoers

#+BEGIN_SRC bash

#+BEGIN_EXAMPLE
%sudoers ALL=(ALL) ALL
%wheel ALL=(ALL) NOPASSWD: ALL

* Add user

#+BEGIN_SRC bash
u=a
g=b,c
sudo useradd -mG $g $u
sudo usermod -aG $u $u
#+END_SRC
